The `config.yml` file is the primary configuration file for Hubbly. It controls all aspects of your hub server including movement features, player settings, world events, and more.

## General Settings

<ParamField path="debug" type="boolean" default="false">
  Enables debug mode for troubleshooting issues. Set to `true` to see detailed console logs.
</ParamField>

<ParamField path="prefix" type="string" default="&e&lHubbly &7&l>&r">
  The plugin prefix used in messages sent to players. Supports color codes and hex colors.
</ParamField>

<ParamField path="check_for_update" type="boolean" default="true">
  Checks for plugin updates on server startup.
</ParamField>

<ParamField path="notify_on_no_update" type="boolean" default="false">
  Send a message if no update is available.
</ParamField>

<ParamField path="default_language" type="string" default="en">
  The default language for plugin messages. Available: `en`, `pt`
</ParamField>

## Database Configuration

<ParamField path="database.enabled" type="boolean" default="true">
  Enable MySQL database storage for player data.
</ParamField>

<ParamField path="database.host" type="string" default="localhost">
  Database host address.
</ParamField>

<ParamField path="database.port" type="integer" default="3306">
  Database port number.
</ParamField>

<ParamField path="database.database" type="string" default="minecraft">
  Database name.
</ParamField>

<ParamField path="database.username" type="string" default="hubbly">
  Database username for authentication.
</ParamField>

<ParamField path="database.password" type="string" default="hubblypassword">
  Database password for authentication.
</ParamField>

<Warning>
  Always use a strong password for your database and never share your credentials.
</Warning>

## Movement Features

### Launchpad

<ParamField path="launchpad.enabled" type="boolean" default="true">
  Enable launchpad functionality.
</ParamField>

<ParamField path="launchpad.type" type="material" default="SLIME_BLOCK">
  The block type that acts as a launchpad.
</ParamField>

<ParamField path="launchpad.power" type="double" default="2">
  Horizontal launch power multiplier.
</ParamField>

<ParamField path="launchpad.power_y" type="double" default="0.5">
  Vertical launch power multiplier.
</ParamField>

<ParamField path="launchpad.cooldown" type="integer" default="2">
  Cooldown in seconds between launchpad uses.
</ParamField>

<ParamField path="launchpad.sound" type="string" default="entity.firework_rocket.blast">
  Sound effect played when using a launchpad.
</ParamField>

### Double Jump

<ParamField path="double_jump.enabled" type="boolean" default="true">
  Enable double jump ability for players.
</ParamField>

<ParamField path="double_jump.power" type="double" default="1.5">
  Horizontal power of the double jump.
</ParamField>

<ParamField path="double_jump.power_y" type="double" default="0.5">
  Vertical power of the double jump.
</ParamField>

<ParamField path="double_jump.cooldown" type="integer" default="2">
  Cooldown in seconds between double jumps.
</ParamField>

<Note>
  Cooldown values must be integers only.
</Note>

## Teleportation

### Anti-Void

<ParamField path="antivoid.enabled" type="boolean" default="true">
  Automatically teleport players to spawn when they fall into the void.
</ParamField>

### Spawn Location

<ParamField path="spawn.world" type="string" default="world">
  World name for the spawn location.
</ParamField>

<ParamField path="spawn.x" type="double" default="0">
  X coordinate of spawn.
</ParamField>

<ParamField path="spawn.y" type="double" default="200">
  Y coordinate of spawn.
</ParamField>

<ParamField path="spawn.z" type="double" default="0">
  Z coordinate of spawn.
</ParamField>

<ParamField path="spawn.yaw" type="float" default="0">
  Yaw rotation of spawn. Auto-rounds to: -135, -90, -45, 0, 45, 90, 135 when using `/setspawn`.
</ParamField>

<ParamField path="spawn.pitch" type="float" default="0">
  Pitch rotation of spawn.
</ParamField>

<ParamField path="spawn.timer" type="integer" default="2">
  Delay in seconds before teleporting to spawn.
</ParamField>

<ParamField path="spawn.sound" type="string" default="entity.firework_rocket.blast">
  Sound played when teleporting to spawn.
</ParamField>

<Note>
  Yaw and pitch will round to specific values when using `/setspawn`, but you can manually override this in the config.
</Note>

## Player Settings

<ParamField path="player.forceinventory" type="boolean" default="true">
  Force players to have a specific inventory on join.
</ParamField>

### Experience Bar

<ParamField path="player.experience.enabled" type="boolean" default="true">
  Enable custom experience level display.
</ParamField>

<ParamField path="player.experience.level" type="integer" default="2025">
  The experience level to display.
</ParamField>

<ParamField path="player.experience.progress" type="float" default="1.0">
  Experience bar progress (0.0 - 1.0). Set to `"YEAR"` to base progress on time elapsed in the year.
</ParamField>

<ParamField path="player.fly.enabled" type="boolean" default="true">
  Allow players to fly in the hub.
</ParamField>

### Join/Leave Messages

<ParamField path="player.join_message.enabled" type="boolean" default="true">

<ParamField path="player.join_message.message" type="string" default="%name% joined">
  Join message format. Use `%name%` for player name.
</ParamField>

<ParamField path="player.leave_message.enabled" type="boolean" default="true">
  Enable custom leave messages.
</ParamField>

<ParamField path="player.leave_message.message" type="string" default="%name% left">
  Leave message format. Use `%name%` for player name.
</ParamField>

### Boss Bar

<ParamField path="player.bossbar.enabled" type="boolean" default="true">
  Enable animated boss bar display.
</ParamField>

<ParamField path="player.bossbar.color" type="string" default="PURPLE">
  Boss bar color. Available: BLUE, GREEN, PINK, PURPLE, RED, WHITE, YELLOW.
</ParamField>

<ParamField path="player.bossbar.animation.change-interval" type="integer" default="40">
  Interval in ticks between text changes (20 ticks = 1 second).
</ParamField>

<ParamField path="player.bossbar.animation.texts" type="array">
  List of texts to cycle through. Supports color codes and hex colors.
  
  ```yaml
  texts:
    - '&6&lWelcome'
    - '<#69b320>&lHello!'
  ```
</ParamField>

## Actions on Join

Define actions that execute when players join the server.

<Accordion title="Available Action Types">
  - **[TITLE]** - Display title to player: `[TITLE] <title>;<subtitle>;<fadeIn>;<stay>;<fadeOut>`
  - **[SOUND]** - Play sound: `[SOUND] <sound_name>`
  - **[FIREWORK]** - Spawn firework: `[FIREWORK] <type>;<red>;<green>;<blue>;<power>`
  - **[ITEM]** - Give item: `[ITEM] <item_id>;<slot>`
</Accordion>

```yaml
actions_on_join:
  - '[TITLE] <#FFDF00>Hubbly;configure me;20;70;20'
  - '[SOUND] block.note_block.chime'
  - '[FIREWORK] BALL;255;0;255;2'
  - '[ITEM] socials;1'
  - '[ITEM] compass;5'
  - '[ITEM] shop;9'
```

## Chat Management

<ParamField path="lock_chat" type="boolean" default="false">
  Lock the chat so only operators can send messages.
</ParamField>

### Blocked Words

<ParamField path="blocked_words.enabled" type="boolean" default="false">
  Enable word filtering.
</ParamField>

<ParamField path="blocked_words.method" type="string" default="">
  Filter method: `STAR` (replace with ****) or `CANCEL` (block message).
</ParamField>

<ParamField path="blocked_words.words" type="array">
  List of words to filter.
  
  ```yaml
  words:
    - 'Fuck'
  ```
</ParamField>

### Blocked Commands

<ParamField path="blocked_commands" type="array">
  List of commands to block for non-operators.
  
  ```yaml
  blocked_commands:
    - 'pl'
    - 'plugins'
    - 'op'
    - 'stop'
    - 'reload'
  ```
</ParamField>

## Announcements

<ParamField path="announcements.enabled" type="boolean" default="true">
  Enable automatic announcements.
</ParamField>

<ParamField path="announcements.interval" type="integer" default="30">
  Interval in seconds between announcements.
</ParamField>

<ParamField path="announcements.global" type="boolean" default="true">
  Send announcements even if the player is in a disabled world.
</ParamField>

<ParamField path="announcements.messages" type="object">
  Define announcement messages. Each message is an array of lines.
  
  ```yaml
  messages:
    message1:
      - '<#FF00FF><center> Test</center>'
      - '<center><#084CFB>This is an announcement</center>'
    message2:
      - 'Test2'
      - 'This is a second announcement'
  ```
</ParamField>

<Note>
  You can use [BirdFlop RGB](https://www.birdflop.com/resources/rgb/) to create cool gradient text!
</Note>

  enabled: true

  power_y: 0.5
  cooldown: 2

playervisibility:
  enabled: true
  visible:
    item: 'LIME_DYE'
    text: '&rPlayers: &aVisible'
  hidden:
    item: 'GRAY_DYE'
    text: '&rPlayers: &cHidden'

antivoid:
  enabled: true

spawn:
  world: "world"
  x: 0
  y: 200
  z: 0
  yaw: 0
  pitch: 0
  timer: 2
  sound: entity.firework_rocket.blast

player:
  forceinventory: true
  experience:
    enabled: true
    level: 2025
    progress: 1.0
  fly:
    enabled: true
  join_message:
    enabled: true
    message: "%name% joined"
  leave_message:
    enabled: true
    message: "%name% left"
  bossbar:
    enabled: true
    color: "PURPLE"
    animation:
      change-interval: 40
      texts:
        - '&6&lWelcome'
        - '<#69b320>&lHello!'

actions_on_join:
  - '[TITLE] <#FFDF00>Hubbly;configure me;20;70;20'
  - '[SOUND] block.note_block.chime'
  - '[FIREWORK] BALL;255;0;255;2'

cancel_events:
  block_place: true
  block_break: true
  interact: true
  damage: true
  weather: true
  item_drop: true
  item_pickup: true
  item_throw: true
  hunger: true
  mob_spawn: true
  block_burn: true
  death_messages: true
  leaf_decay: true
  projectiles: false
  animal_eat: false

anti_world_download:
  enabled: true
  notify_admin: true

lock_chat: false

blocked_words:
  enabled: false
  method: ''

    - 'Example'

blocked_commands:
  - 'pl'
  - 'plugins'

  enabled: true
  interval: 30
  global: true
  messages:
    message1:
      - '<#FF00FF><center>Welcome!</center>'
